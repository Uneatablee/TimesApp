
project(qt_implementation VERSION 0.1 LANGUAGES CXX)

set(PROJECT_SOURCES
        core/main.cpp
        core/mainwindow.cpp
        core/mainwindow.ui
        core/mainwindow.h

        view/QtViewStyles.hpp
        view/QtUserView.cpp
        view/QtUserView.hpp
)

find_package(Qt6 REQUIRED COMPONENTS Widgets)

qt_standard_project_setup()

qt_add_executable(qt_implementation ${PROJECT_SOURCES})

target_link_libraries(qt_implementation PRIVATE Qt6::Widgets)
target_link_libraries(qt_implementation INTERFACE day_planner_functions)

set_target_properties(qt_implementation PROPERTIES
    WIN32_EXECUTABLE ON
    MACOSX_BUNDLE ON
)
